# Hanzi H5 MVP Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a mobile-first H5 MVP that lets a parent complete one 5-minute, story-led literacy session per card, with lightweight local progress tracking.

**Architecture:** Create a new frontend app in `app/` using Vite + React + TypeScript. Keep all curated card content in local typed data, render a storybook-style learning flow with no backend, and persist `completed / favorite / recent` state in `localStorage`.

**Tech Stack:** Vite, React, TypeScript, React Router, Vitest, React Testing Library, CSS variables, `localStorage`, Web Speech API (`speechSynthesis`)

```ts
// app/src/content/cards/index.ts
import { beiCard } from './bei'

const cards = [beiCard]

export function listCards() {
  return cards
}

export function getDailyCard() {
  return cards[0]
}

export function getCardBySlug(slug: string) {
  return cards.find((card) => card.slug === slug)
}

export function findCardByQuery(query: string) {
  const normalized = query.trim().toLowerCase()
  return cards.find(
    (card) =>
      card.slug === normalized ||
      card.character === query.trim() ||
      card.pinyin.toLowerCase() === normalized,
  )
}
```

```tsx
// app/src/features/home/HomePage.tsx
import { FormEvent, useState } from 'react'
import { useNavigate } from 'react-router-dom'
import { findCardByQuery, getDailyCard } from '../../content/cards'

export function HomePage() {
  const navigate = useNavigate()
  const dailyCard = getDailyCard()
  const [query, setQuery] = useState('')

  function openQuery(event: FormEvent) {
    event.preventDefault()
    const card = findCardByQuery(query)
    if (card) navigate(`/cards/${card.slug}`)
  }

  return (
    <section className="hero-stack">
      <article className="hero-card">
        <p className="eyebrow">今天的故事字</p>
        <h1>{dailyCard.character}</h1>
        <p>{dailyCard.heroLine}</p>
        <button type="button" onClick={() => navigate(`/cards/${dailyCard.slug}`)}>
          开始今天这张卡
        </button>
      </article>

      <form className="panel-card" onSubmit={openQuery}>
        <label htmlFor="card-query">搜一个字</label>
        <input
          id="card-query"
          value={query}
          onChange={(event) => setQuery(event.target.value)}
        />
        <button type="submit">打开这个字卡</button>
      </form>
    </section>
  )
}
```

```ts
// app/src/lib/audio/speak.ts
export function speakText(text: string) {
  if (!('speechSynthesis' in window)) return false

  window.speechSynthesis.cancel()
  const utterance = new SpeechSynthesisUtterance(text)
  utterance.lang = 'zh-CN'
  window.speechSynthesis.speak(utterance)
  return true
}
```

```tsx
// app/src/features/card/CardPage.tsx
import { useMemo } from 'react'
import { useNavigate, useParams } from 'react-router-dom'
import { getCardBySlug } from '../../content/cards'
import { speakText } from '../../lib/audio/speak'

export function CardPage() {
  const { slug = '' } = useParams()
  const navigate = useNavigate()
  const card = useMemo(() => getCardBySlug(slug), [slug])

  if (!card) {
    return (
      <section className="panel-card">
        <p>没有找到这张字卡。</p>
        <button type="button" onClick={() => navigate('/')}>
          回到首页
        </button>
      </section>
    )
  }

  return (
    <section className="story-page">
      <article className="storybook-hero">
        <p className="eyebrow">故事画面</p>
        <div className="scene-block">{card.storyScene}</div>
        <h1>{card.character}</h1>
        <p>{card.heroLine}</p>
        <div className="button-row">
          <button type="button" onClick={() => speakText(card.storyText)}>
            听这个故事
          </button>
          <button type="button">我来慢慢讲</button>
        </div>
      </article>

      <article className="panel-card">
        <h2>给家长的小提示</h2>
        <p>{card.parentPrompt}</p>
      </article>

      <article className="panel-card">
        <h2>往下读一读</h2>
        <p>{card.words.join(' / ')}</p>
        {card.sentences.map((sentence) => (
          <p key={sentence}>{sentence}</p>
        ))}
      </article>

      <article className="panel-card">
        <h2>最后一起玩一下</h2>
        <p>{card.activityPrompt}</p>
      </article>

      <button type="button">今天这张读完了</button>
    </section>
  )
}
```

- [ ] **Step 3: Implement progress storage and wire it into the UI**

```ts
// app/src/lib/progress/store.ts
const STORAGE_KEY = 'hanzi-h5-progress'

export interface ProgressEntry {
  completed: boolean
  favorite: boolean
  lastOpenedAt: string | null
}

export type ProgressMap = Record<string, ProgressEntry>

export function readProgress(): ProgressMap {
  const raw = localStorage.getItem(STORAGE_KEY)
  return raw ? JSON.parse(raw) : {}
}

function writeProgress(progress: ProgressMap) {
  localStorage.setItem(STORAGE_KEY, JSON.stringify(progress))
}

export function markCompleted(slug: string) {
  const current = readProgress()
  current[slug] = {
    completed: true,
    favorite: current[slug]?.favorite ?? false,
    lastOpenedAt: new Date().toISOString(),
  }
  writeProgress(current)
}

export function toggleFavorite(slug: string) {
  const current = readProgress()
  current[slug] = {
    completed: current[slug]?.completed ?? false,
    favorite: !(current[slug]?.favorite ?? false),
    lastOpenedAt: new Date().toISOString(),
  }
  writeProgress(current)
}
```

```tsx
// app/src/features/profile/ProfilePage.tsx
import { listCards } from '../../content/cards'
import { readProgress } from '../../lib/progress/store'

export function ProfilePage() {
  const progress = readProgress()
  const completed = Object.entries(progress).filter(([, entry]) => entry.completed)
  const favorites = Object.entries(progress).filter(([, entry]) => entry.favorite)

  return (
    <section className="panel-card">
      <h1>我的</h1>
      <p>学过 {completed.length} 张</p>
      <p>收藏 {favorites.length} 张</p>
      <ul className="card-list">
        {listCards()
          .filter((card) => progress[card.slug])
          .map((card) => (
            <li key={card.slug}>
              {card.character} · {progress[card.slug].favorite ? '已收藏' : '已读过'}
            </li>
          ))}
      </ul>
    </section>
  )
}
```
